### 3. Backend XP System
typescript
// backend/trpc/routes/user/add-xp/route.ts

// XP Calculation Formula:
function calculateLevel(totalXP: number): number {
  let level = 1;
  let xpNeeded = 0;
  
  while (xpNeeded <= totalXP) {
    xpNeeded += 1000  level;  // Level 1 needs 1000, Level 2 needs 2000, etc.
    if (xpNeeded <= totalXP) {
      level++;
    }
  }
  
  return level;
}

// Database fields needed:
// - users.total_xp (integer)
// - users.level (integer)
```

## Design System
- Colors: Dark theme (#0F1520 background, #4ADE80 accent green)
- Gradients: Background uses teal-to-dark gradient (#0F1520 → #134E4A → #0F1520)
- Animations: React Native Animated API for star particles
- Layout: Bottom sheet modal (90% height, rounded top corners)

## Integration Pattern
```typescript
// Usage in profile or header:
const [showLevelModal, setShowLevelModal] = useState(false);

<TouchableOpacity onPress={() => setShowLevelModal(true)}>
  <LevelBadge level={user.level} currentXP={user.total_xp} />
</TouchableOpacity>

<LevelDetailsModal
  visible={showLevelModal}
  onClose={() => setShowLevelModal(false)}
  level={user.level}
  currentXP={user.total_xp}
  userId={user.id}
  onTaskStart={(taskId) => {
    // Handle task navigation
    if (taskId === 'upload-video') router.push('/create');
  }}
/>
```
