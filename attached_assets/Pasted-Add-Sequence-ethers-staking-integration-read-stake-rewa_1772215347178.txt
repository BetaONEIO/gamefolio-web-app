// Minimal ABI (only what we need)
const STAKING_ABI = [
  "function stakes(address) view returns (uint256 amount, uint256 rewardDebt, uint256 lastClaim)",
  "function totalStaked() view returns (uint256)",
  "function earned(address) view returns (uint256)",
  "function pendingRewards(address) view returns (uint256)",
  "function claimRewards()",
];

// Format helper
const fmt = (v: ethers.BigNumberish) => Number(ethers.formatUnits(v, TOKEN_DECIMALS));

/**
 * 1) READ ONLY: staked amount + earned + pending
 *    Uses public RPC provider (no wallet popups, minimal UX)
 */
export async function getStakingSnapshot(userAddress: string) {
  const provider = new ethers.JsonRpcProvider(SKALE_RPC_URL);
  const staking = new ethers.Contract(STAKING_ADDRESS, STAKING_ABI, provider);

  const [stakeInfo, earned, pending, totalStaked] = await Promise.all([
    staking.stakes(userAddress),
    staking.earned(userAddress),
    staking.pendingRewards(userAddress),
    staking.totalStaked(),
  ]);

  // stakeInfo: { amount, rewardDebt, lastClaim }
  return {
    staked: fmt(stakeInfo.amount),
    rewardDebt: fmt(stakeInfo.rewardDebt),
    lastClaim: Number(stakeInfo.lastClaim),
    earned: fmt(earned),
    pending: fmt(pending),
    totalStaked: fmt(totalStaked),
  };
}

export async function claimStakingRewards() {
  const signer = await getSequenceSigner();
  const staking = new ethers.Contract(STAKING_ADDRESS, STAKING_ABI, signer);

  const tx = await staking.claimRewards();
  // Minimal UX: show a single “Claiming…” state until confirmed
  const receipt = await tx.wait();

  return {
    hash: receipt?.hash ?? tx.hash,
    status: receipt?.status ?? 1,
  };
}
